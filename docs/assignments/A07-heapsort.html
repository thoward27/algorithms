The Algorithm

def heapsort(A):
   # Create a heap.
   Heap h
   # Push all elements to the heap.
   for x in A:
      h.push(x)
   # Refill the original array with the in-order elements.
   for i in range(0, len(A)):
      A[i] = h.pop()
